<!-- Glow Logic Script -->
<script>
const sphere = document.getElementById('centerSphere');
const apexes = document.querySelectorAll('.apex');

apexes.forEach(apex => {
    apex.addEventListener('mouseenter', () => {
        const step = apex.dataset.step;

        const glowLevels = {
            1: '0 0 35px rgba(0,150,255,0.9)',
            2: '0 0 45px rgba(0,150,255,1)',
            3: '0 0 55px rgba(0,150,255,1)',
            4: '0 0 65px rgba(0,150,255,1)',
            5: '0 0 75px rgba(0,150,255,1)'
        };

        sphere.style.boxShadow = glowLevels[step];
        sphere.style.background =
            'radial-gradient(circle, rgba(0,150,255,0.9), rgba(0,0,0,0.85))';
    });

    apex.addEventListener('mouseleave', () => {
        sphere.style.boxShadow = '0 0 25px rgba(0,150,255,0.7)';
        sphere.style.background =
            'radial-gradient(circle, rgba(0,150,255,0.6), rgba(0,0,0,0.8))';
    });
});
</script>
